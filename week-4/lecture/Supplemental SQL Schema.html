DROP TABLE IF EXISTS WatchHistory;
DROP TABLE IF EXISTS Subscriptions;
DROP TABLE IF EXISTS Content;
DROP TABLE IF EXISTS Users;


CREATE TABLE Users (
    user_id NUMBER(10) PRIMARY KEY,
    name VARCHAR2(100) NOT NULL,
    country VARCHAR2(50),
    signup_date DATE NOT NULL
);

CREATE TABLE Content (
    content_id NUMBER(10) PRIMARY KEY,
    title VARCHAR2(200) NOT NULL,
    genre VARCHAR2(50),
    release_year NUMBER(4)
);

CREATE TABLE Subscriptions (
    sub_id NUMBER(10) PRIMARY KEY,
    user_id NUMBER(10) NOT NULL,
    plan VARCHAR2(20) NOT NULL,
    monthly_fee NUMBER(6,2) NOT NULL,
    start_date DATE NOT NULL,

    CONSTRAINT fk_subscriptions_user
        FOREIGN KEY (user_id) REFERENCES Users(user_id),
    CONSTRAINT chk_subscriptions_plan
        CHECK (plan IN ('Basic', 'Standard', 'Premium', 'Student'))
);

CREATE TABLE WatchHistory (
    watch_id NUMBER(10) PRIMARY KEY,
    user_id NUMBER(10) NOT NULL,
    content_id NUMBER(10) NOT NULL,
    watch_date DATE NOT NULL,
    watch_time_minutes NUMBER(4) NOT NULL,
    rating NUMBER(2),

    CONSTRAINT fk_watchhistory_user
        FOREIGN KEY (user_id) REFERENCES Users(user_id),
    CONSTRAINT fk_watchhistory_content
        FOREIGN KEY (content_id) REFERENCES Content(content_id),
    CONSTRAINT chk_watchhistory_rating
        CHECK (rating BETWEEN 1 AND 5 OR rating IS NULL),
    CONSTRAINT chk_watchhistory_minutes
        CHECK (watch_time_minutes >= 0)
);
